// root/node_modules/restfuljs/foo.js

var request = require("request");

exports.get = function( req, res ) {

  var args = {
    "method": "GET",
    "url": "http://wedata.net/databases/JapanAPI/items_all.json",
  };
  request( args, function( error, response, body ){
    res.send( body )
  });

  return false;

};
